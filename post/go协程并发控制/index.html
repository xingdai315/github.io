const (
    conNum = 4	//cpu cores
)
func main() {
    wg := &sync.WaitGroup{}
    ch := make(chan bool, conNum)
    var list map[string]string
    err := someApi(api_url, &list)
    if err != nil {
        log.Fatalln(err)
    }

    for k, v := range list {
        wg.Add(1)
        ch <- true
        go func(wg *sync.WaitGroup, id string, url string) {
            err := doSomeThing(id, url)
            if err != nil {
                log.Println(err)
            }
            <-ch
            defer wg.Done()
        }(wg, k, v)
    }
    wg.Wait()
    defer log.Println("process done.")
}
